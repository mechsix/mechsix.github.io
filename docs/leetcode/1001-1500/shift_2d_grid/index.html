1260. Shift 2D Grid

https://leetcode.com/problems/shift-2d-grid/

Pythonâ€‹

Offset 1 per runâ€‹

  • Time: O(mnk)
  • Space: O(1)
class Solution:
    def shiftGrid(self, grid: List[List[int]], k: int) -> List[List[int]]:
        m = len(grid)
        n = len(grid[0])

        for i in range(k):
            last = grid[-1][-1]
            for r in range(m):
                tmp = grid[r][-1]
                for c in range(n-1, 0, -1):
                    grid[r][c] = grid[r][c-1]
                grid[r][0] = last
                last = tmp

        return grid
