1446. Consecutive Characters

class Solution:
    def maxPower(self, s: str) -> int:
        length = len(s)

        if length <= 1:
            return 1

        result = 1
        max_result = 1

        for index in range(1, length):
            if s[index] == s[index-1]:
                result += 1
            else:
                result = 1
            max_result = max(result, max_result)

        return max_result
